	int arr[3][3] = {{3, 2, 5}, {8, 1, 7}, {6, 4, 9}};

	int minOfMax = INT_MAX;

	for (int i = 0; i < 3; i++) {
	int rowMax = arr[i][0];

	for (int j = 1; j < 3; j++) {
	if (arr[i][j] > rowMax) {
	rowMax = arr[i][j];
	}
	}

	if (rowMax < minOfMax) {
	minOfMax = rowMax;
	}
	}

	cout << "Минимальный среди максимальных элементов строки: " << minOfMax << endl;
