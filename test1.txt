// Нахождение минимального среди максимальных элементов строки
	// Исходный двумерный массив
	int arr[3][3] = {{3, 2, 5}, {8, 1, 7}, {6, 4, 9}};

	// Переменная для хранения минимального среди максимальных
	int minOfMax = INT_MAX;

	// Проходим по каждой строке
	for (int i = 0; i < 3; i++) {
	int rowMax = arr[i][0]; // Предполагаем первый элемент строки максимальным

	// Ищем максимальный элемент в текущей строке
	for (int j = 1; j < 3; j++) {
	if (arr[i][j] > rowMax) {
	rowMax = arr[i][j];
	}
	}

	// Сравниваем найденный максимум с минимальным среди максимумов
	if (rowMax < minOfMax) {
	minOfMax = rowMax;
	}
	}

	// Выводим результат
	cout << "Минимальный среди максимальных элементов строки: " << minOfMax << endl;
